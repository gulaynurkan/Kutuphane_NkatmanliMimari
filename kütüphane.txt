use master;
create database kutuphane;
use kutuphane;

create table kullanicilar(
kullanici_id int identity,
kullanici_adi varchar(20),
kullanici_sifre varchar(20),
constraint const_pkkey_kullanicilar primary key (kullanici_adi)
);

create table diller(
d_id int identity,
d_adi varchar(20),
constraint const_pkkey_diller primary key (d_id)
);

create table kitaptürleri(
kt_id int identity,
kt_kodu varchar(6) constraint const_kt_kodu check(kt_kodu like('KTK-[0-9][0-9]')),--KTK-00
kt_adi varchar(30),
constraint const_pkkey_kitaptürleri primary key (kt_kodu)
);

create table yazarlar(
y_id int identity,
y_kodu varchar(8) constraint const_y_kodu check(y_kodu like('YK-[0-9][0-9][0-9][0-9][0-9]')),--YK-00000
y_adi varchar(50),
y_soyadi varchar(50),
constraint const_pkkey_yazarlar primary key (y_kodu)
);

create table yayinevi(
ye_id int identity,
ye_firmakodu varchar(9) constraint const_ye_firmakodu check(ye_firmakodu like('YEFK-[0-9][0-9][0-9][0-9]')),--YEFK-0000
ye_firmaadi varchar(50),
ye_firmatelI varchar(16),
ye_firmatelII varchar(16),
ye_firmafaxno varchar(16),
ye_firmaemail varchar(60),
ye_firmawebadresi varchar(100),
ye_firmaadresi varchar(500),
constraint const_pkkey_yayinevi primary key (ye_firmakodu)
);

create table kitaplar(
ktp_id int identity,
ktp_kodu uniqueidentifier,
ktp_ISBN varchar(13) unique not null,
ktp_adi varchar(150),
ktp_kt_kodu varchar(6) foreign key references kitaptürleri(kt_kodu),
ktp_y_kodu varchar(8) foreign key references yazarlar(y_kodu),
ktp_sayfasayisi int,
ktp_ye_firmakodu varchar(9) foreign key references yayinevi(ye_firmakodu),
ktp_basimtarihi date,
constraint const_pkkey_kitaplar primary key (ktp_kodu)
);

create table kitapfiyatlari(
ktp_kodu uniqueidentifier foreign key references kitaplar(ktp_kodu),
ktp_fiyati money,
ktp_gunlukkiralamafiyati money,
);


create table uyeler(
u_id int identity,
u_kodu varchar(8) constraint const_u_kodu check(u_kodu like('UK-[0-9][0-9][0-9][0-9][0-9]')),--UK-00000
u_adi varchar(30),
u_soyadi varchar(30), 
u_dogumtarihi date,
u_tel varchar(16),
u_email varchar(50), 
u_adres varchar(500),
constraint const_pkkey_uyeler primary key (u_kodu)
);

create table kiralama(
k_id int identity,
k_kodu varchar(10) constraint const_k_kodu check(k_kodu like('KK-[0-9][0-9][0-9][0-9][0-9][0-9][0-9]')),--KK-0000000
k_ktp_kodu uniqueidentifier foreign key references kitaplar(ktp_kodu),
k_baslangictarihi date,
k_bitistarihi date,
k_veristarihi date,
k_u_kodu varchar(8) foreign key references uyeler(u_kodu),
k_fiyati money,
k_d_kodu int foreign key references diller(d_id),
k_notu varchar(500),
k_bitisdurumu bit,
constraint const_pkkey_kiralama primary key (k_kodu)
);
